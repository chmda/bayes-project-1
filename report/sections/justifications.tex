    \begin{itemize}
    
        \item 
        $\pi(\beta_k | ...)$= $ \propto \pi(\beta_k) \prod_{i,j} \pi(Y_{ij}|...)$ \\

        $\sim \mathcal{N}((\sum_{i,j}{ \nu_{ij}(Y_{ij} - \delta_{ij}) \tau_{ij})}(\tau + \sum_{i,j}{ \nu_{i,j}^2 \tau_{ij}})^{-1} , (\tau + \sum_{i,j} \nu_{i,j}^2 \tau_{ij})^{-1})$
